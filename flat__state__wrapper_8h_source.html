292  void add_data(const Entity_kind on_what, const std::string var_name, std::shared_ptr<std::vector<T>> data) {
293  // if we have seen this entity before - check size match, else store this size
294  auto sziter = entity_size_map_.find(on_what);
295  if (sziter != entity_size_map_.end()) {
296  if (sziter->second != data->size()) {
297  throw std::runtime_error(std::string("variable ")+var_name+" has incompatible size on add");
298  }
299  } else {
300  entity_size_map_[on_what] = data->size();
301  }
302 
303  // store data and update internal book-keeping
304  pair_t pair(var_name, on_what);
305  auto iter = name_map_.find(pair);
306  if (iter == name_map_.end()) { // have not seen this entity-name combo before, add data
307  state_.push_back(data);
308  name_map_[pair] = state_.size() - 1;
309  entities_.push_back(on_what);
310  entity_map_[var_name] = on_what;
311  entity_size_map_[on_what] = data->size();
312  } else { // have seen the entity-name combo already, replace data. already checked size.
313  std::copy(data->begin(), data->end(), state_[iter->second]->begin());
314  }
315  }
