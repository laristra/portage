386  // SEARCH
387  Portage::vector<std::vector<unsigned int>> candidates(numTargetPts);
388 
389  // Get an instance of the desired search algorithm type which is expected
390  // to be a functor with an operator() of the right form
391 
392  gettimeofday(&begin_timeval, 0);
393 
394  // search boxes around source points are of zero dimension but
395  // those around target points are determined by particle
396  // smoothing lengths
397 
398  // code below does not work with facted weights
399  std::shared_ptr<vector<Point<Dim>>> sourceExtents;
400  std::shared_ptr<vector<Point<Dim>>> targetExtents;
401  targetExtents = std::make_shared<vector<Point<Dim>>>(numTargetPts);
402  if (weight_center_ == Portage::Meshfree::Gather) {
403  for (int i = 0; i < numTargetPts; i++) {
404  if (geom_types_[i] == Weight::FACETED) {
405  throw std::runtime_error("FACETED geometry is not available here");
406  }
407  {Point<Dim> pt=(*targetExtents)[i];
408  std::vector<std::vector<double>> vv=smoothing_lengths_[i];
409  pt=Point<Dim>(vv[0]); (*targetExtents)[i]=pt;}
410  }
411  }
412 
413  int numSourcePts = source_swarm_.num_particles();
414  sourceExtents = std::make_shared<vector<Point<Dim>>>(numSourcePts);
415  if (weight_center_ == Portage::Meshfree::Scatter) {
416  for (int i = 0; i < numSourcePts; i++) {
417  if (geom_types_[i] == Weight::FACETED) {
418  throw std::runtime_error("FACETED geometry is not available here");
419  }
420  {Point<Dim> pt=(*sourceExtents)[i];

482  //amh: ?? add back accuracy output statement??
483  if (comm_rank == 0) std::cout << "Remapping swarm variable " <<
484  src_varnames[i] <<
485  " to variable " << trg_varnames[i] <<
486  std::endl;
487 
488  estimateFunctor.set_variable(src_varnames[i]);
489 
490  // This populates targetField with the values returned by the
491  // remapper operator
492 
493  // ***************** NOTE NOTE NOTE NOTE ********************
494  // THE CURRENT SWARM_STATE DOES NOT HAVE AN OPERATOR TO RETURN
495  // A RAW POINTER TO THE DATA. INSTEAD IT RETURNS THIS REQUIRES
496  // A SPECIFIC TYPE OF THE SWARM_STATE CLASS. THIS IS A BAD
497  // IDEA BECAUSE IT FORCES ALL OTHER SWARM_STATE CLASSES TO
498  // HAVE THIS SAME DEFINITION. IT ALSO LEADS TO THE UGLY USAGE

503 
504  target_state_.get_field(trg_varnames[i], target_field_shared_ptr);

506

508  target_swarm_.end(Entity_kind::PARTICLE, Entity_type::PARALLEL_OWNED),

512 
513  gettimeofday(&end_timeval, 0);
514  timersub(&end_timeval, &begin_timeval, &diff_timeval);
515  tot_seconds_interp = diff_timeval.tv_sec + 1.0E-6*diff_timeval.tv_usec;
516 
517  tot_seconds = tot_seconds_dist + tot_seconds_srch +
